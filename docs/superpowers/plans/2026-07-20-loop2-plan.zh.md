# Loop 2 实施计划

> **给 AI 执行者：** 必须使用 superpowers:subagent-driven-development（推荐）或 superpowers:executing-plans 逐任务执行本计划。步骤使用复选框（`- [ ]`）语法追踪进度。

**目标：** 构建轮询服务器和触发脚本，自动 pick up `ready/` 中已批准的 plan，以 full-auto 模式运行 Codex 完成实现 → 验证 → 开 PR，支持一次 retry，两次失败后 escalate 到 `failed/`。

**架构：** `loop2-server.js` 按 cron 计划 poll `ready/`，对每个发现的 ticket 并行 spawn `loop-2-trigger.sh <ticket>`。bash 脚本获取 lockfile，运行 `codex --approval-mode full-auto`，读取 Codex 写入的 `result.json`，将 ticket 移到 `done/` 或 `failed/`。

**技术栈：** Node.js 18+、node-cron、child_process.spawn、bash、codex CLI

## 全局约束

- Node.js >= 18
- 仅工作日 9-5 PST：cron `0 9-17 * * 1-5`，时区 `America/Los_Angeles`
- 所有操作记录到 stdout，带 ISO 时间戳
- 错误时不崩溃——记录日志并继续
- lockfile 过期阈值：2 小时（与 Loop 1 相同）
- Codex 必须以 `--approval-mode full-auto` 调用
- server 启动前 `done/` 和 `failed/` 目录必须存在

- [ ] **步骤 2：编写 `loop-2-trigger.sh`**

```bash
#!/bin/bash
set -euo pipefail

TICKET="${1:?Usage: loop-2-trigger.sh <ticket>}"
BASE=~/workspace/dev-loop
READY="$BASE/ready/$TICKET"
LOCKFILE="$BASE/ready/$TICKET.lock"
NOW=$(date -u +%Y-%m-%dT%H:%M:%SZ)
SESSION="${NOW//[^0-9]/}"
SESSION="${SESSION: -8}"

log() { echo "[$(date -u +%Y-%m-%dT%H:%M:%SZ)] [loop2] [$TICKET] $*"; }

# lockfile 检查
if [[ -f "$LOCKFILE" ]]; then
  ACQUIRED=$(python3 -c "import json; print(json.load(open('$LOCKFILE'))['acquired'])")
  ACQUIRED_S=$(date -j -f "%Y-%m-%dT%H:%M:%SZ" "$ACQUIRED" +%s 2>/dev/null)
  NOW_S=$(date +%s)
  AGE_H=$(( (NOW_S - ACQUIRED_S) / 3600 ))
  if [[ $AGE_H -lt 2 ]]; then
    log "Skipped — lockfile active (age: ${AGE_H}h)"
    exit 0
  fi
  log "Stale lockfile detected (age: ${AGE_H}h), overwriting"
fi

echo "{\"session\": \"$SESSION\", \"acquired\": \"$NOW\"}" > "$LOCKFILE"
log "Lockfile acquired"

CODEX_PROMPT="Read ~/workspace/dev-loop/ready/$TICKET/plan.md.
Use superpowers:subagent-driven-development to implement all tasks.
When complete, run superpowers:verification-before-completion.
Then run superpowers:finishing-a-development-branch.
Work in ~/workspace/mortgage-graphify.
When done, write ~/workspace/dev-loop/ready/$TICKET/result.json:
- On success: {\"status\": \"success\", \"pr_url\": \"<url>\"}
- On failure: {\"status\": \"failed\", \"reason\": \"<reason>\"}"

run_codex() {
  local attempt=$1
  log "Starting Codex attempt $attempt"
  rm -f "$READY/result.json"
  codex exec "$CODEX_PROMPT" || true
}

read_result() {
  if [[ ! -f "$READY/result.json" ]]; then
    echo "result.json missing"
    return 1
  fi
  python3 -c "import json,sys; d=json.load(open('$READY/result.json')); sys.exit(0 if d['status']=='success' else 1)" 2>/dev/null
}

read_reason() {
  if [[ ! -f "$READY/result.json" ]]; then
    echo "result.json missing"
  else
    python3 -c "import json; d=json.load(open('$READY/result.json')); print(d.get('reason', d.get('pr_url', 'unknown')))"
  fi
}

# 第一次尝试
run_codex 1
REASON1=$(read_reason)
if read_result; then
  log "Attempt 1 succeeded — PR: $REASON1"
  mv "$READY" "$BASE/done/$TICKET"
  rm -f "$LOCKFILE"
  log "Moved to done/"
  exit 0
fi
log "Attempt 1 failed: $REASON1"

# 第二次尝试（retry）
run_codex 2
REASON2=$(read_reason)
if read_result; then
  log "Attempt 2 succeeded — PR: $REASON2"
  mv "$READY" "$BASE/done/$TICKET"
  rm -f "$LOCKFILE"
  log "Moved to done/"
  exit 0
fi
log "Attempt 2 failed: $REASON2"

# 两次均失败——escalate
mkdir -p "$BASE/failed/$TICKET"
cat > "$BASE/failed/$TICKET/error.md" <<EOF
# $TICKET — 两次尝试均失败

Attempt 1: $REASON1
Attempt 2: $REASON2
时间戳: $NOW
EOF
mv "$READY"/* "$BASE/failed/$TICKET/" 2>/dev/null || true
rmdir "$READY" 2>/dev/null || true
rm -f "$LOCKFILE"
log "Escalated to failed/"
exit 1
```

### Task 3：实现 `loop2-server.js`

**文件：**
- 新建：`~/workspace/dev-loop/loop2-server.js`

**接口：**
- 消费：`~/workspace/dev-loop/ready/` 目录
- 消费：`~/workspace/dev-loop/ready/<ticket>.lock`（spawn 前检查）
- 产出：为每个 ticket spawn `loop-2-trigger.sh` 子进程

- [ ] **步骤 1：编写 `loop2-server.js`**

```js
import cron from 'node-cron'
import fs from 'fs'
import path from 'path'
import { spawn } from 'child_process'
import { fileURLToPath } from 'url'
import 'dotenv/config'

const __dirname = path.dirname(fileURLToPath(import.meta.url))
const READY_DIR = path.join(__dirname, 'ready')
const TRIGGER = path.join(process.env.HOME, '.claude/skills/loop-2-trigger/loop-2-trigger.sh')

function log(msg) {
  console.log(`[${new Date().toISOString()}] [loop2] ${msg}`)
}

function isLockActive(ticket) {
  const lockfile = path.join(READY_DIR, `${ticket}.lock`)
  if (!fs.existsSync(lockfile)) return false
  try {
    const { acquired } = JSON.parse(fs.readFileSync(lockfile, 'utf8'))
    const ageH = (Date.now() - new Date(acquired).getTime()) / 3_600_000
    return ageH < 2
  } catch {
    return false
  }
}

function findPendingTickets() {
  if (!fs.existsSync(READY_DIR)) return []
  return fs.readdirSync(READY_DIR, { withFileTypes: true })
    .filter(d => d.isDirectory() && fs.existsSync(path.join(READY_DIR, d.name, 'plan.md')))
    .map(d => d.name)
}

function spawnTrigger(ticket) {
  log(`Spawning trigger for ${ticket}`)
  const proc = spawn('bash', [TRIGGER, ticket], {
    detached: true,
    stdio: 'ignore',
  })
  proc.on('error', err => log(`Spawn error for ${ticket}: ${err.message}`))
  proc.unref()
}

function poll() {
  log('Poll start')
  const tickets = findPendingTickets()
  log(`Tickets found: ${tickets.length}`)

  for (const ticket of tickets) {
    if (isLockActive(ticket)) {
      log(`Skipped (lockfile active): ${ticket}`)
      continue
    }
    spawnTrigger(ticket)
  }

  log('Poll end')
}

// 工作日 9-5 PST，每小时整点
cron.schedule('0 9-17 * * 1-5', poll, { timezone: 'America/Los_Angeles' })
log('loop2-server.js started — polling weekdays 9-17 PST')

// 启动时立即执行一次
poll()
```

# Loop 2 Trigger

Loop 2 由 `loop2-server.js` 通过 bash 触发，不是 Claude Code CronCreate。

本 skill 仅作文档用途——记录 Codex prompt 和预期行为，便于维护。

## `loop-2-trigger.sh` 做的事

1. 获取 lockfile（`ready/<ticket>.lock`，2 小时过期阈值）
2. 运行 `codex --approval-mode full-auto`
3. 读取 Codex 写入的 `ready/<ticket>/result.json`
4. 成功 → 移动 ticket 到 `done/<ticket>/`
5. 失败 → retry 一次；第二次失败 → 移到 `failed/<ticket>/`，写 `error.md`

## Codex prompt

```
Read ~/workspace/dev-loop/ready/<ticket>/plan.md.
Use superpowers:subagent-driven-development to implement all tasks.
When complete, run superpowers:verification-before-completion.
Then run superpowers:finishing-a-development-branch.
Work in ~/workspace/mortgage-graphify.
When done, write ~/workspace/dev-loop/ready/<ticket>/result.json:
- On success: {"status": "success", "pr_url": "<url>"}
- On failure: {"status": "failed", "reason": "<reason>"}
```

## 人工介入点

- **成功**：PR 在 GitHub 上开出——正常 review 和 merge。
- **失败**：ticket 出现在 `failed/`——查看 `error.md`，修复 plan 或 ticket，移回 `ready/` 重试。
